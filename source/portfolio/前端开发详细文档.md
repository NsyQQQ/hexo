```typescript
// Http.ts
export class Http {
  private baseUrl: string = 'https://your-domain.com/api';
  
  async request<T>(url: string, options: RequestOptions): Promise<T> {
    const token = localStorage.getItem('token');
    
    const response = await fetch(this.baseUrl + url, {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        'Authorization': token ? `Bearer ${token}` : '',
        ...options.headers,
      },
    });
    
    const result = await response.json();
    if (result.code !== 0) {
      throw new Error(result.msg || '请求失败');
    }
    return result.data;
  }
  
  get<T>(url: string): Promise<T> {
    return this.request<T>(url, { method: 'GET' });
  }
  
  post<T>(url: string, data?: any): Promise<T> {
    return this.request<T>(url, {
      method: 'POST',
      body: JSON.stringify(data),
    });
  }
}

export const http = new Http();
```

### WebSocket 连接封装

```typescript
// Socket.ts
export class Socket {
  private socket: WebSocket;
  private url: string = 'wss://your-domain.com/ws';
  private handlers: Map<string, Function[]> = new Map();
  private reconnectTimer: number = null;
  private heartbeatTimer: number = null;
  
  connect() {
    return new Promise<void>((resolve, reject) => {
      this.socket = new WebSocket(this.url);
      
      this.socket.onopen = () => {
        console.log('WebSocket connected');
        this.startHeartbeat();
        resolve();
      };
      
      this.socket.onmessage = (event) => {
        const message = JSON.parse(event.data);
        this.handleMessage(message);
      };
      
      this.socket.onerror = (error) => {
        console.error('WebSocket error:', error);
        reject(error);
      };
      
      this.socket.onclose = () => {
        console.log('WebSocket closed');
        this.stopHeartbeat();
        this.reconnect();
      };
    });
  }
  
  // 发送消息
  send(type: string, data?: any) {
    if (this.socket.readyState === WebSocket.OPEN) {
      this.socket.send(JSON.stringify({ type, data }));
    }
  }
  
  // 监听消息
  on(type: string, handler: Function) {
    if (!this.handlers.has(type)) {
      this.handlers.set(type, []);
    }
    this.handlers.get(type).push(handler);
  }
  
  // 移除监听
  off(type: string, handler: Function) {
    const handlers = this.handlers.get(type);
    if (handlers) {
      const index = handlers.indexOf(handler);
      if (index > -1) {
        handlers.splice(index, 1);
      }
    }
  }
  
  // 处理消息
  private handleMessage(message: any) {
    const handlers = this.handlers.get(message.type);
    if (handlers) {
      handlers.forEach(handler => handler(message.data));
    }
  }
  
  // 心跳
  private startHeartbeat() {
    this.heartbeatTimer = setInterval(() => {
      this.send('heartbeat', { timestamp: Date.now() });
    }, 30000);
  }
  
  private stopHeartbeat() {
    if (this.heartbeatTimer) {
      clearInterval(this.heartbeatTimer);
      this.heartbeatTimer = null;
    }
  }
  
  // 重连
  private reconnect() {
    if (this.reconnectTimer) return;
    
    this.reconnectTimer = setTimeout(() => {
      this.reconnectTimer = null;
      this.connect();
    }, 3000);
  }
}

export const socket = new Socket();
```

### 牌型判断

```typescript
// CardLogic.ts
export class CardLogic {
  // 牌型定义
  static readonly CARD_TYPE = {
    HIGH_CARD: 1,      // 高牌
    PAIR: 2,           // 对子
    TWO_PAIR: 3,       // 两对
    THREE_OF_A_KIND: 4, // 三条
    STRAIGHT: 5,       // 顺子
    FLUSH: 6,           // 同花
    FULL_HOUSE: 7,      // 葫芦
    FOUR_OF_A_KIND: 8,  // 四条
    STRAIGHT_FLUSH: 9,  // 同花顺
    ROYAL_FLUSH: 10,   // 皇家同花顺
  };
  
  // 判断牌型
  static getCardType(cards: number[]): number {
    // 排序
    const sorted = [...cards].sort((a, b) => this.getCardValue(b) - this.getCardValue(a));
    
    // 判断各种牌型
    if (this.isRoyalFlush(sorted)) return this.CARD_TYPE.ROYAL_FLUSH;
    if (this.isStraightFlush(sorted)) return this.CARD_TYPE.STRAIGHT_FLUSH;
    if (this.isFourOfAKind(sorted)) return this.CARD_TYPE.FOUR_OF_A_KIND;
    if (this.isFullHouse(sorted)) return this.CARD_TYPE.FULL_HOUSE;
    if (this.isFlush(sorted)) return this.CARD_TYPE.FLUSH;
    if (this.isStraight(sorted)) return this.CARD_TYPE.STRAIGHT;
    if (this.isThreeOfAKind(sorted)) return this.CARD_TYPE.THREE_OF_A_KIND;
    if (this.isTwoPair(sorted)) return this.CARD_TYPE.TWO_PAIR;
    if (this.isPair(sorted)) return this.CARD_TYPE.PAIR;
    
    return this.CARD_TYPE.HIGH_CARD;
  }
  
  // 比牌
  static compare(cards1: number[], cards2: number[]): number {
    const type1 = this.getCardType(cards1);
    const type2 = this.getCardType(cards2);
    
    if (type1 !== type2) {
      return type1 > type2 ? 1 : -1;
    }
    
    // 同牌型比大小
    return this.compareHighCard(cards1, cards2);
  }
}
```

```typescript
export class LoginUI extends Component {
  async onLoad() {
    // 检查是否已登录
    const token = localStorage.getItem('token');
    if (token) {
      this.enterHall();
      return;
    }
  }
  
  async onLoginBtnClick() {
    // 获取微信授权
    const loginBtn = this.loginBtn.getComponent(Button);
    loginBtn.interactable = false;
    
    try {
      // 微信登录
      const code = await this.wxLogin();
      
      // 请求服务器登录
      const data = await http.post<LoginResponse>('/login', { code });
      
      // 保存token
      localStorage.setItem('token', data.token);
      localStorage.setItem('userId', data.userId.toString());
      
      // 进入大厅
      this.enterHall();
    } catch (error) {
      console.error('登录失败:', error);
      loginBtn.interactable = true;
    }
  }

    return new Promise((resolve, reject) => {
      // @ts-ignore
      wx.login({
        success: (res) => resolve(res.code),
        fail: reject,
      });
    });
  }

    sceneManager.loadScene('Hall');
  }
}
```

```typescript
export class CardAnimation {
  static async dealCards(cards: number[], positions: Vec3[]): Promise<void> {
    const promises = cards.map((card, index) => {
      return this.moveCard(card, positions[index]);
    });
    
    await Promise.all(promises);
  }
  
  private static moveCard(card: Node, targetPos: Vec3): Promise<void> {
    return new Promise(resolve => {
      // 创建移动动作
      const moveAction = tween(card.position)
        .to(0.3, targetPos, {
          easing: 'sineOut',
        })
        .call(resolve)
        .start();
    });
  }
  
  // 翻牌动画
  static async flipCard(card: Node): Promise<void> {
    return new Promise(resolve => {
      // 翻牌动作
      tween(card)
        .to(0.15, { scaleX: 0 })
        .call(() => {
          // 切换牌面
        })
        .to(0.15, { scaleX: 1 })
        .call(resolve)
        .start();
    });
  }
}
```
