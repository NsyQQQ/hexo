## 1. 复杂度分析

### 1.1 时间复杂度

时间复杂度表示算法执行时间与输入规模的关系。常用记号：

- **O(1)**：常数时间
- **O(log n)**：对数时间
- **O(n)**：线性时间
- **O(n log n)**：线性对数时间
- **O(n²)**：平方时间
- **O(2^n)**：指数时间

### 1.2 空间复杂度

空间复杂度表示算法所需内存空间与输入规模的关系。

## 2. 常见数据结构

### 2.1 数组

数组是最基本的数据结构，提供快速的随机访问。

```javascript
// 数组基本操作
class ArrayList {
    constructor() {
        this.data = [];
        this.size = 0;
    }

    add(element) {
        this.data.push(element);
        this.size++;
    }

    get(index) {
        return this.data[index];
    }

    remove(index) {
        return this.data.splice(index, 1)[0];
    }
}
```

### 2.2 链表

链表是一种线性数据结构，每个节点包含数据和指向下一个节点的指针。

```javascript
class ListNode {
    constructor(val) {
        this.val = val;
        this.next = null;
    }
}

class LinkedList {
    constructor() {
        this.head = null;
        this.size = 0;
    }

    addAtHead(val) {
        const node = new ListNode(val);
        node.next = this.head;
        this.head = node;
        this.size++;
    }

    reverse() {
        let prev = null;
        let curr = this.head;
        
        while (curr) {
            const next = curr.next;
            curr.next = prev;
            prev = curr;
            curr = next;
        }
        
        this.head = prev;
    }
}
```

```javascript
// 有效括号
function isValid(s) {
    const stack = new Stack();
    const map = { ')': '(', ']': '[', '}': '{' };
    
    for (const char of s) {
        if (char in map) {
            if (stack.pop() !== map[char]) return false;
        } else {
            stack.push(char);
        }
    }
    
    return stack.items.length === 0;
}
```

```javascript
// 两数之和
function twoSum(nums, target) {
    const map = new Map();
    
    for (let i = 0; i < nums.length; i++) {
        const complement = target - nums[i];
        if (map.has(complement)) {
            return [map.get(complement), i];
        }
        map.set(nums[i], i);
    }
}
```

### 2.5 树

二叉树是每个节点最多有两个子节点的数据结构。

```javascript
class TreeNode {
    constructor(val) {
        this.val = val;
        this.left = null;
        this.right = null;
    }
}

// 二叉搜索树
class BST {
    constructor() {
        this.root = null;
    }

    insert(val) {
        const node = new TreeNode(val);
        
        if (!this.root) {
            this.root = node;
            return;
        }
        
        let curr = this.root;
        while (true) {
            if (val < curr.val) {
                if (!curr.left) { curr.left = node; break; }
                curr = curr.left;
            } else {
                if (!curr.right) { curr.right = node; break; }
                curr = curr.right;
            }
        }
    }

    // 层序遍历
    levelOrder() {
        if (!this.root) return [];
        const result = [];
        const queue = [this.root];
        
        while (queue.length) {
            const level = [];
            const len = queue.length;
            
            for (let i = 0; i < len; i++) {
                const node = queue.shift();
                level.push(node.val);
                if (node.left) queue.push(node.left);
                if (node.right) queue.push(node.right);
            }
            result.push(level);
        }
        return result;
    }
}
```

### 2.6 图

图由顶点和边组成。

```javascript
class Graph {
    constructor() {
        this.adjacencyList = new Map();
    }

    addVertex(v) {
        if (!this.adjacencyList.has(v)) {
            this.adjacencyList.set(v, []);
        }
    }

    addEdge(v1, v2) {
        this.addVertex(v1);
        this.addVertex(v2);
        this.adjacencyList.get(v1).push(v2);
        this.adjacencyList.get(v2).push(v1);
    }

    // BFS
    bfs(start) {
        const visited = new Set();
        const queue = [start];
        const result = [];
        
        while (queue.length) {
            const vertex = queue.shift();
            if (!visited.has(vertex)) {
                visited.add(vertex);
                result.push(vertex);
                for (const neighbor of this.adjacencyList.get(vertex)) {
                    queue.push(neighbor);
                }
            }
        }
        return result;
    }

    // DFS
    dfs(start) {
        const visited = new Set();
        const result = [];
        
        const dfsHelper = (v) => {
            visited.add(v);
            result.push(v);
            for (const neighbor of this.adjacencyList.get(v)) {
                if (!visited.has(neighbor)) dfsHelper(neighbor);
            }
        };
        
        dfsHelper(start);
        return result;
    }
}
```

## 3. 常见算法

### 3.1 排序算法

```javascript
// 快速排序
function quickSort(arr, left = 0, right = arr.length - 1) {
    if (left >= right) return;
    
    const pivot = arr[right];
    let i = left - 1;
    
    for (let j = left; j < right; j++) {
        if (arr[j] <= pivot) {
            i++;
            [arr[i], arr[j]] = [arr[j], arr[i]];
        }
    }
    
    [arr[i + 1], arr[right]] = [arr[right], arr[i + 1]];
    const pi = i + 1;
    
    quickSort(arr, left, pi - 1);
    quickSort(arr, pi + 1, right);
    return arr;
}

// 归并排序
function mergeSort(arr) {
    if (arr.length <= 1) return arr;
    
    const mid = Math.floor(arr.length / 2);
    const left = mergeSort(arr.slice(0, mid));
    const right = mergeSort(arr.slice(mid));
    
    return merge(left, right);
}

function merge(left, right) {
    const result = [];
    let i = 0, j = 0;
    
    while (i < left.length && j < right.length) {
        if (left[i] <= right[j]) result.push(left[i++]);
        else result.push(right[j++]);
    }
    
    return result.concat(left.slice(i)).concat(right.slice(j));
}
```

### 3.2 查找算法

```javascript
// 二分查找
function binarySearch(arr, target) {
    let left = 0, right = arr.length - 1;
    
    while (left <= right) {
        const mid = Math.floor((left + right) / 2);
        
        if (arr[mid] === target) return mid;
        else if (arr[mid] < target) left = mid + 1;
        else right = mid - 1;
    }
    
    return -1;
}
```

### 3.3 递归与回溯

```javascript
// 全排列
function permute(nums) {
    const result = [];
    
    const backtrack = (path, used) => {
        if (path.length === nums.length) {
            result.push([...path]);
            return;
        }
        
        for (let i = 0; i < nums.length; i++) {
            if (used[i]) continue;
            path.push(nums[i]);
            used[i] = true;
            backtrack(path, used);
            path.pop();
            used[i] = false;
        }
    };
    
    backtrack([], []);
    return result;
}

// 子集
function subsets(nums) {
    const result = [];
    
    const backtrack = (start, path) => {
        result.push([...path]);
        
        for (let i = start; i < nums.length; i++) {
            path.push(nums[i]);
            backtrack(i + 1, path);
            path.pop();
        }
    };
    
    backtrack(0, []);
    return result;
}
```

### 3.4 动态规划

```javascript
// 斐波那契
function fibonacci(n) {
    if (n <= 1) return n;
    const dp = new Array(n + 1).fill(0);
    dp[1] = 1;
    for (let i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }
    return dp[n];
}

// 背包问题
function knapSack(weights, values, capacity) {
    const n = weights.length;
    const dp = new Array(n + 1).fill(0).map(() => new Array(capacity + 1).fill(0));
    
    for (let i = 1; i <= n; i++) {
        for (let w = 0; w <= capacity; w++) {
            if (weights[i - 1] <= w) {
                dp[i][w] = Math.max(dp[i - 1][w], dp[i - 1][w - weights[i - 1]] + values[i - 1]);
            } else {
                dp[i][w] = dp[i - 1][w];
            }
        }
    }
    return dp[n][capacity];
}

// 最长递增子序列
function lis(nums) {
    const dp = new Array(nums.length).fill(1);
    
    for (let i = 1; i < nums.length; i++) {
        for (let j = 0; j < i; j++) {
            if (nums[i] > nums[j]) {
                dp[i] = Math.max(dp[i], dp[j] + 1);
            }
        }
    }
    return Math.max(...dp);
}
```

### 3.5 算法技巧

```javascript
// 双指针 - 有序数组去重
function removeDuplicates(nums) {
    if (nums.length === 0) return 0;
    let slow = 0;
    for (let fast = 1; fast < nums.length; fast++) {
        if (nums[fast] !== nums[slow]) {
            slow++;
            nums[slow] = nums[fast];
        }
    }
    return slow + 1;
}

// 滑动窗口 - 无重复字符最长子串
function lengthOfLongestSubstring(s) {
    const charIndex = new Map();
    let maxLength = 0, left = 0;
    
    for (let right = 0; right < s.length; right++) {
        if (charIndex.has(s[right]) && charIndex.get(s[right]) >= left) {
            left = charIndex.get(s[right]) + 1;
        }
        charIndex.set(s[right], right);
        maxLength = Math.max(maxLength, right - left + 1);
    }
    return maxLength;
}

// 位运算 - 统计1的个数
function countBits(n) {
    let count = 0;
    while (n) {
        n &= n - 1;
        count++;
    }
    return count;
}
```
